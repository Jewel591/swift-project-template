```swift
import Foundation

enum ServerConfig {

    private static let defaultURL = "https://api.cluster-fluster.com"
    private static let storageKey = "happy_custom_server_url"

    static var serverURL: String {
        if let custom = UserDefaults.standard.string(forKey: storageKey) {
            return custom
        }
        return defaultURL
    }

    static var isUsingCustomServer: Bool {
        serverURL != defaultURL
    }

    static func setCustomURL(_ url: String?) {
        if let url = url?.trimmingCharacters(in: .whitespaces), !url.isEmpty {
            UserDefaults.standard.set(url, forKey: storageKey)
        } else {
            UserDefaults.standard.removeObject(forKey: storageKey)
        }
    }

    struct ValidationResult {
        let isValid: Bool
        let error: String?
    }

    static func validate(_ urlString: String) -> ValidationResult {
        guard !urlString.trimmingCharacters(in: .whitespaces).isEmpty else {
            return ValidationResult(isValid: false, error: "Server URL cannot be empty")
        }
        guard let url = URL(string: urlString.trimmingCharacters(in: .whitespaces)),
              let scheme = url.scheme,
              ["http", "https"].contains(scheme),
              url.host != nil else {
            return ValidationResult(isValid: false, error: "Invalid URL format")
        }
        return ValidationResult(isValid: true, error: nil)
    }
}
```

```swift
import Foundation
import Security

struct AuthCredentials: Codable, Equatable {
    let token: String
    let secret: String  // base64url-encoded master secret
}

enum CredentialStoreError: Error {
    case encodingFailed
    case keychainError(OSStatus)
}

enum CredentialStore {

    private static let service = "com.happycoder.auth"
    private static let account = "auth_credentials"

    static func getCredentials() async throws -> AuthCredentials? {
        let query: [String: Any] = [
            kSecClass as String: kSecClassGenericPassword,
            kSecAttrService as String: service,
            kSecAttrAccount as String: account,
            kSecReturnData as String: true,
            kSecMatchLimit as String: kSecMatchLimitOne,
        ]

        var result: AnyObject?
        let status = SecItemCopyMatching(query as CFDictionary, &result)

        if status == errSecItemNotFound {
            return nil
        }
        guard status == errSecSuccess, let data = result as? Data else {
            throw CredentialStoreError.keychainError(status)
        }

        return try JSONDecoder().decode(AuthCredentials.self, from: data)
    }

    static func setCredentials(_ credentials: AuthCredentials) async throws {
        let data = try JSONEncoder().encode(credentials)

        // Delete existing item first
        await removeCredentials()

        let query: [String: Any] = [
            kSecClass as String: kSecClassGenericPassword,
            kSecAttrService as String: service,
            kSecAttrAccount as String: account,
            kSecValueData as String: data,
            kSecAttrAccessible as String: kSecAttrAccessibleWhenUnlockedThisDeviceOnly,
        ]

        let status = SecItemAdd(query as CFDictionary, nil)
        guard status == errSecSuccess else {
            throw CredentialStoreError.keychainError(status)
        }
    }

    static func removeCredentials() async {
        let query: [String: Any] = [
            kSecClass as String: kSecClassGenericPassword,
            kSecAttrService as String: service,
            kSecAttrAccount as String: account,
        ]
        SecItemDelete(query as CFDictionary)
    }
}
```

```swift
import Foundation

enum SecretKeyBackupError: Error {
    case invalidFormat
    case invalidKeyLength(got: Int, expected: Int)
    case noValidCharacters
}

enum SecretKeyBackup {

    private static let base32Alphabet = "ABCDEFGHIJKLMNOPQRSTUVWXYZ234567"

    // MARK: - Base32 (RFC 4648)

    static func bytesToBase32(_ data: Data) -> String {
        let alphabet = Array(base32Alphabet)
        var result = ""
        var buffer = 0
        var bufferLength = 0

        for byte in data {
            buffer = (buffer << 8) | Int(byte)
            bufferLength += 8

            while bufferLength >= 5 {
                bufferLength -= 5
                result.append(alphabet[(buffer >> bufferLength) & 0x1F])
            }
        }

        if bufferLength > 0 {
            result.append(alphabet[(buffer << (5 - bufferLength)) & 0x1F])
        }

        return result
    }

    static func base32ToBytes(_ base32: String) throws -> Data {
        // Normalize: uppercase, fix common mistakes, remove non-base32 chars
        var normalized = base32.uppercased()
            .replacingOccurrences(of: "0", with: "O")
            .replacingOccurrences(of: "1", with: "I")
            .replacingOccurrences(of: "8", with: "B")
            .replacingOccurrences(of: "9", with: "G")

        let cleaned = normalized.filter { base32Alphabet.contains($0) }

        guard !cleaned.isEmpty else {
            throw SecretKeyBackupError.noValidCharacters
        }

        var bytes = Data()
        var buffer = 0
        var bufferLength = 0

        for char in cleaned {
            guard let value = base32Alphabet.firstIndex(of: char) else {
                throw SecretKeyBackupError.invalidFormat
            }
            let index = base32Alphabet.distance(from: base32Alphabet.startIndex, to: value)

            buffer = (buffer << 5) | index
            bufferLength += 5

            if bufferLength >= 8 {
                bufferLength -= 8
                bytes.append(UInt8((buffer >> bufferLength) & 0xFF))
            }
        }

        return bytes
    }

    // MARK: - Format / Parse

    /// Format a base64url-encoded secret key for human-readable backup.
    /// Output: "XXXXX-XXXXX-XXXXX-..." (Base32 groups of 5, joined by dashes)
    static func formatForBackup(_ base64urlKey: String) -> String {
        let bytes = Base64Utils.decodeURL(base64urlKey)
        let base32 = bytesToBase32(bytes)

        var groups: [String] = []
        var i = base32.startIndex
        while i < base32.endIndex {
            let end = base32.index(i, offsetBy: 5, limitedBy: base32.endIndex) ?? base32.endIndex
            groups.append(String(base32[i..<end]))
            i = end
        }

        return groups.joined(separator: "-")
    }

    /// Parse a human-readable backup key back to base64url format.
    static func parseBackup(_ formatted: String) throws -> String {
        let bytes = try base32ToBytes(formatted)

        guard bytes.count == 32 else {
            throw SecretKeyBackupError.invalidKeyLength(got: bytes.count, expected: 32)
        }

        return Base64Utils.encodeURL(bytes)
    }

    /// Check if a string is a valid secret key (either base64url or formatted).
    static func isValid(_ key: String) -> Bool {
        if key.contains("-") || key.count > 50 {
            guard let parsed = try? parseBackup(key) else { return false }
            return Base64Utils.decodeURL(parsed).count == 32
        }
        return Base64Utils.decodeURL(key).count == 32
    }
}
```

```swift
import Foundation
import Sodium

struct ChallengeResult {
    let challenge: Data   // 32 bytes random
    let signature: Data   // 64 bytes Ed25519
    let publicKey: Data   // 32 bytes Ed25519
}

enum AuthChallenge {

    private static let sodium = Sodium()

    /// Create a signed challenge for authentication.
    /// Mirrors: authChallenge(secret) in authChallenge.ts
    static func create(secret: Data) -> ChallengeResult {
        // Derive Ed25519 keypair from master secret
        let seedBytes = Bytes(secret)
        let keypair = sodium.sign.keyPair(seed: seedBytes)!

        // Generate random 32-byte challenge
        let challenge = Data(sodium.randomBytes.buf(length: 32)!)

        // Sign the challenge
        let signature = sodium.sign.signature(
            message: Bytes(challenge),
            secretKey: keypair.secretKey
        )!

        return ChallengeResult(
            challenge: challenge,
            signature: Data(signature),
            publicKey: Data(keypair.publicKey)
        )
    }

    /// Verify a signature (for testing).
    static func verify(signature: Data, message: Data, publicKey: Data) -> Bool {
        sodium.sign.verify(
            message: Bytes(message),
            publicKey: Bytes(publicKey),
            signature: Bytes(signature)
        )
    }
}
```

```swift
import Foundation

enum HappyAPIError: Error {
    case invalidURL
    case httpError(statusCode: Int, data: Data?)
    case decodingFailed
    case networkError(Error)
}

enum HappyAPIClient {

    // MARK: - Auth Endpoints

    /// POST /v1/auth — Challenge-response authentication
    /// Mirrors: authGetToken.ts
    static func buildAuthRequest(challenge: Data, signature: Data, publicKey: Data) -> URLRequest {
        let body: [String: String] = [
            "challenge": Base64Utils.encode(challenge),
            "signature": Base64Utils.encode(signature),
            "publicKey": Base64Utils.encode(publicKey),
        ]
        return jsonPost(path: "/v1/auth", body: body)
    }

    /// POST /v1/auth/account/request — QR auth initiation/polling
    /// Mirrors: authQRStart.ts + authQRWait.ts
    static func buildAccountRequest(publicKey: Data) -> URLRequest {
        let body = ["publicKey": Base64Utils.encode(publicKey)]
        return jsonPost(path: "/v1/auth/account/request", body: body)
    }

    /// POST /v1/auth/account/response — Approve QR auth (desktop side)
    /// Mirrors: authAccountApprove.ts
    static func buildAccountResponse(token: String, publicKey: Data, response: Data) -> URLRequest {
        let body: [String: String] = [
            "publicKey": Base64Utils.encode(publicKey),
            "response": Base64Utils.encode(response),
        ]
        var request = jsonPost(path: "/v1/auth/account/response", body: body)
        request.setValue("Bearer \(token)", forHTTPHeaderField: "Authorization")
        return request
    }

    // MARK: - Execute

    /// Execute a URLRequest and decode JSON response.
    static func execute<T: Decodable>(_ request: URLRequest) async throws -> T {
        let (data, response) = try await URLSession.shared.data(for: request)

        if let httpResponse = response as? HTTPURLResponse,
           !(200..<300).contains(httpResponse.statusCode) {
            throw HappyAPIError.httpError(statusCode: httpResponse.statusCode, data: data)
        }

        return try JSONDecoder().decode(T.self, from: data)
    }

    /// Execute a URLRequest, ignoring response body.
    static func executeVoid(_ request: URLRequest) async throws {
        let (data, response) = try await URLSession.shared.data(for: request)

        if let httpResponse = response as? HTTPURLResponse,
           !(200..<300).contains(httpResponse.statusCode) {
            throw HappyAPIError.httpError(statusCode: httpResponse.statusCode, data: data)
        }
    }

    // MARK: - Helpers

    private static func jsonPost(path: String, body: [String: String]) -> URLRequest {
        let url = URL(string: ServerConfig.serverURL + path)!
        var request = URLRequest(url: url)
        request.httpMethod = "POST"
        request.setValue("application/json", forHTTPHeaderField: "Content-Type")
        request.httpBody = try? JSONSerialization.data(withJSONObject: body)
        return request
    }
}
```

```swift
import Foundation
import Sodium

enum QRAuthFlowError: Error {
    case decryptionFailed
    case serverError(String)
    case cancelled
}

enum QRAuthFlow {

    private static let sodium = Sodium()

    // MARK: - Key Generation

    /// Generate a random X25519 keypair for QR auth.
    /// Mirrors: generateAuthKeyPair() in authQRStart.ts
    static func generateKeyPair() -> BoxKeyPair {
        let seed = Data(sodium.randomBytes.buf(length: 32)!)
        return BoxCrypto.generateKeyPairFromSeed(seed)
    }

    /// The QR code payload: base64-encoded public key.
    /// Desktop scans this to encrypt the master secret.
    static func qrPayload(keypair: BoxKeyPair) -> String {
        Base64Utils.encode(keypair.publicKey)
    }

    // MARK: - Response Decryption

    /// Decrypt the server's auth response to extract master secret.
    /// The server encrypts masterSecret using Box with our publicKey.
    /// Mirrors: decryptBox() call in authQRWait.ts
    static func decryptAuthResponse(encryptedResponse: String, secretKey: Data) throws -> Data {
        let encryptedData = Base64Utils.decode(encryptedResponse)
        guard let decrypted = BoxCrypto.decryptOrNil(encryptedData, recipientSecretKey: secretKey) else {
            throw QRAuthFlowError.decryptionFailed
        }
        return decrypted
    }

    // MARK: - Server Communication

    /// Response from POST /v1/auth/account/request
    struct AccountRequestResponse: Decodable {
        let state: String        // "pending" | "authorized"
        let token: String?       // only when authorized
        let response: String?    // only when authorized — encrypted master secret
    }

    /// Initiate QR auth by registering public key with server.
    /// Mirrors: authQRStart() in authQRStart.ts
    static func start(keypair: BoxKeyPair) async throws -> Bool {
        let request = HappyAPIClient.buildAccountRequest(publicKey: keypair.publicKey)
        do {
            try await HappyAPIClient.executeVoid(request)
            return true
        } catch {
            return false
        }
    }

    /// Poll server until auth is approved. Returns credentials on success.
    /// Mirrors: authQRWait() in authQRWait.ts
    static func waitForApproval(
        keypair: BoxKeyPair,
        shouldCancel: @escaping () -> Bool = { false },
        onProgress: ((Int) -> Void)? = nil
    ) async throws -> AuthCredentials {
        var dots = 0

        while !shouldCancel() {
            let request = HappyAPIClient.buildAccountRequest(publicKey: keypair.publicKey)
            let response: AccountRequestResponse = try await HappyAPIClient.execute(request)

            if response.state == "authorized",
               let token = response.token,
               let encryptedResponse = response.response {
                let masterSecret = try decryptAuthResponse(
                    encryptedResponse: encryptedResponse,
                    secretKey: keypair.secretKey
                )
                return AuthCredentials(
                    token: token,
                    secret: Base64Utils.encodeURL(masterSecret)
                )
            }

            onProgress?(dots)
            dots += 1
            try await Task.sleep(for: .seconds(1))
        }

        throw QRAuthFlowError.cancelled
    }

    // MARK: - Token Refresh

    /// Get a fresh token using master secret (challenge-response).
    /// Mirrors: authGetToken() in authGetToken.ts
    static func getToken(secret: Data) async throws -> String {
        let result = AuthChallenge.create(secret: secret)
        let request = HappyAPIClient.buildAuthRequest(
            challenge: result.challenge,
            signature: result.signature,
            publicKey: result.publicKey
        )

        struct TokenResponse: Decodable {
            let token: String
        }

        let response: TokenResponse = try await HappyAPIClient.execute(request)
        return response.token
    }
}
```

```swift
import XCTest
@testable import TemplateProject

final class AuthIntegrationTests: XCTestCase {

    func testFullAuthFlowOffline() throws {
        // 1. Mobile generates keypair for QR code
        let keypair = QRAuthFlow.generateKeyPair()
        XCTAssertEqual(keypair.publicKey.count, 32)

        // 2. QR payload is base64 of public key
        let qrPayload = QRAuthFlow.qrPayload(keypair: keypair)
        XCTAssertFalse(qrPayload.isEmpty)

        // 3. Desktop side: generate master secret and encrypt it
        let masterSecret = Data(repeating: 0x42, count: 32)
        let encryptedSecret = try BoxCrypto.encrypt(masterSecret, recipientPublicKey: keypair.publicKey)
        let encryptedBase64 = Base64Utils.encode(encryptedSecret)

        // 4. Mobile decrypts to get master secret
        let decryptedSecret = try QRAuthFlow.decryptAuthResponse(
            encryptedResponse: encryptedBase64,
            secretKey: keypair.secretKey
        )
        XCTAssertEqual(decryptedSecret, masterSecret)

        // 5. Use master secret for challenge-response auth
        let challenge = AuthChallenge.create(secret: decryptedSecret)
        XCTAssertEqual(challenge.challenge.count, 32)
        XCTAssertEqual(challenge.signature.count, 64)
        XCTAssertEqual(challenge.publicKey.count, 32)

        // 6. Verify the signature (server would do this)
        let isValid = AuthChallenge.verify(
            signature: challenge.signature,
            message: challenge.challenge,
            publicKey: challenge.publicKey
        )
        XCTAssertTrue(isValid)

        // 7. Same master secret → same Ed25519 public key (deterministic)
        let challenge2 = AuthChallenge.create(secret: decryptedSecret)
        XCTAssertEqual(challenge.publicKey, challenge2.publicKey)

        // 8. Secret key backup format works
        let base64url = Base64Utils.encodeURL(decryptedSecret)
        let formatted = SecretKeyBackup.formatForBackup(base64url)
        XCTAssertTrue(formatted.contains("-"))
        let restored = try SecretKeyBackup.parseBackup(formatted)
        XCTAssertEqual(Base64Utils.decodeURL(restored), decryptedSecret)
    }
}
```
